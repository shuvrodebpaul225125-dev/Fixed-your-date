    <script>
        let selectedDate = '';
        let selectedDayName = '';
        let selectedFoods = [];

        // Set minimum date to today
        const today = new Date().toISOString().split('T')[0];
        document.getElementById('datePicker').setAttribute('min', today);

        function nextStep(stepNumber) {
            document.querySelectorAll('.step').forEach(step => step.classList.remove('active'));
            document.getElementById(`step${stepNumber}`).classList.add('active');
        }

        // Trick "No" button functionality (playful)
        function moveNoButton() {
            const noBtn = document.getElementById('noBtn');
            const x = Math.random() * (window.innerWidth - noBtn.offsetWidth - 50);
            const y = Math.random() * (window.innerHeight - noBtn.offsetHeight - 50);
            
            noBtn.style.position = 'absolute';
            noBtn.style.left = `${x}px`;
            noBtn.style.top = `${y}px`;
        }

        // Live day updating on choice
        function updateDayName() {
            const dateInput = document.getElementById('datePicker').value;
            const dayDisplay = document.getElementById('dayDisplay');
            
            if (dateInput) {
                const dateObj = new Date(dateInput);
                const options = { weekday: 'long' };
                selectedDayName = new Intl.DateTimeFormat('en-US', options).format(dateObj);
                selectedDate = dateInput;
                dayDisplay.innerText = `That's a beautiful ${selectedDayName}!`;
            } else {
                dayDisplay.innerText = '';
            }
        }

        function goToFoodStep() {
            if (!selectedDate) {
                alert('Please pick a date first! 😉');
                return;
            }
            nextStep(3);
        }

        function toggleFood(element, foodName) {
            element.classList.toggle('selected');
            if (selectedFoods.includes(foodName)) {
                selectedFoods = selectedFoods.filter(item => item !== foodName);
            } else {
                selectedFoods.push(foodName);
            }
        }

        function goToFinalStep() {
            if (selectedFoods.length === 0) {
                alert('Please pick at least one food option! 🍰');
                return;
            }

            // Populate summary
            document.getElementById('finalDate').innerText = selectedDate;
            document.getElementById('finalDay').innerText = selectedDayName;
            document.getElementById('finalFood').innerText = selectedFoods.join(', ');

            nextStep(4);
        }
    </script>
